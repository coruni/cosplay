  ```ts
  export async function cacheDelete(pattern: string): Promise<void> {
    try {
      let cursor = '0';
      do {
        const [next, keys] = await redis.scan(
          cursor,
          'MATCH',
          pattern,
          'COUNT',
          200
        );
        cursor = next;
        if (keys.length > 0) {
          await redis.del(...keys);
        }
      } while (cursor !== '0');
    } catch {
      // Fail silently — cache is optional
    }
  }
  ```

#### 任务 3.1：validateToken 改 SHA-256 比较（P2-1）
- **文件**：[src/lib/auth.ts](file:///f:/project/cosplay/src/lib/auth.ts)
- **改动**：
  ```ts
  export function validateToken(token: string): boolean {
    const valid = getAdminToken();
    const a = crypto.createHash('sha256').update(token).digest();
    const b = crypto.createHash('sha256').update(valid).digest();
    // 长度恒定 32 字节，timingSafeEqual 不会因长度差异抛错
    return crypto.timingSafeEqual(a, b);
  }
  ```

  ```ts
  export async function getRelatedGalleries(
    gallery: Gallery,
    limit = 4,
    showNsfw = false
  ): Promise<Gallery[]> {
    const where: Prisma.GalleryWhereInput = {
      id: { not: gallery.id },
      categories: { hasSome: gallery.categories },
    };
    // 显式 rating 过滤，避免 OR 里的 rating 被顶层覆盖
    where.rating = showNsfw ? gallery.rating : 'sfw';
    const rows = await prisma.gallery.findMany({

      take: limit,
      orderBy: { viewCount: 'desc' },
    });
    return rows.map(toGallery);
  }
  ```
